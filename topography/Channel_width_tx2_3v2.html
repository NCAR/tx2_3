Set up output files#

today = date.today()

path_out = path_in

print('channel width file : ',file_out_chan)

fmt_out = "{0:s}, {1:8.2f}, {2:8.2f}, {3:8.2f}, {4:8.2f}, {5:10.1f} ! {6:s}\n"
